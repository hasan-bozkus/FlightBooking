    {

        return ("#4B5563", "#FFFFFF", "#4B5563");

@section Scripts {
<script>

    var PER_PAGE    = 8;
    var currentPage = 1;

    function allRows() {
        return Array.from(document.querySelectorAll('#tableBody tr[data-flight-no]'));
    }

    function filterTable() {
        currentPage = 1;
        applyFiltersAndRender();
    }

    function applyFiltersAndRender() {
        var q       = (document.getElementById('searchInput').value || '').toLowerCase().trim();
        var status  = document.getElementById('statusFilter').value;
        var airline = document.getElementById('airlineFilter').value;

        var rows    = allRows();

        var visible = rows.filter(function (row) {
            var flightNo   = row.dataset.flightNo   || '';
            var depAirport = row.dataset.depAirport || '';
            var arrAirport = row.dataset.arrAirport || '';
            var rowAirline = row.dataset.airline    || '';
            var rowStatus  = row.dataset.status     || '';

            var matchQ       = !q      || [flightNo, depAirport, arrAirport, rowAirline].some(function (v) { return v.includes(q); });
            var matchStatus  = !status  || rowStatus  === status;
            var matchAirline = !airline || rowAirline === airline;

            row.style.display = 'none';
            return matchQ && matchStatus && matchAirline;
        });

        var total      = visible.length;
        var totalPages = Math.max(1, Math.ceil(total / PER_PAGE));
        if (currentPage > totalPages) currentPage = totalPages;

        var from  = total === 0 ? 0 : (currentPage - 1) * PER_PAGE + 1;
        var to    = Math.min(currentPage * PER_PAGE, total);
        var slice = visible.slice(from - 1, to);

        slice.forEach(function (row) { row.style.display = ''; });

        document.getElementById('rowCount').textContent   = total + ' uçuş';
        document.getElementById('footerInfo').textContent = from + '–' + to + ' / ' + total + ' uçuş';

        renderPagination(totalPages);
    }

    function renderPagination(totalPages) {
        var ul   = document.getElementById('paginationList');
        var html = '';

        html += '<li class="' + (currentPage === 1 ? 'disabled' : '') + '">'
              + '<span onclick="' + (currentPage > 1 ? 'goPage(currentPage-1)' : '') + '"'
              + ' style="cursor:' + (currentPage === 1 ? 'default' : 'pointer') + '">'
              + '<i class="bi bi-chevron-left" style="font-size:11px;"></i></span></li>';

        buildPageRange(currentPage, totalPages).forEach(function (p) {
            if (p === '...') {
                html += '<li><span style="cursor:default;border:none;color:var(--gray-300);">…</span></li>';
            } else {
                html += '<li class="' + (p === currentPage ? 'active' : '') + '">'
                      + '<a onclick="goPage(' + p + ')">' + p + '</a></li>';
            }
        });

        html += '<li class="' + (currentPage === totalPages ? 'disabled' : '') + '">'
              + '<span onclick="' + (currentPage < totalPages ? 'goPage(currentPage+1)' : '') + '"'
              + ' style="cursor:' + (currentPage === totalPages ? 'default' : 'pointer') + '">'
              + '<i class="bi bi-chevron-right" style="font-size:11px;"></i></span></li>';

        ul.innerHTML = html;
    }

    function buildPageRange(current, total) {
        if (total <= 7) {
            var arr = [];
            for (var i = 1; i <= total; i++) arr.push(i);
            return arr;
        }
        var pages = [1];
        if (current > 3) pages.push('...');
        for (var p = Math.max(2, current - 1); p <= Math.min(total - 1, current + 1); p++) pages.push(p);
        if (current < total - 2) pages.push('...');
        pages.push(total);
        return pages;
    }

    function goPage(p) {
        currentPage = p;
        applyFiltersAndRender();
    }

    /* Delete modal */
    function confirmDelete(flightId, flightNo) {
        document.getElementById('deleteFlightNo').textContent = flightNo;
        document.getElementById('deleteForm').action = '/Admin/Flights/Delete/' + flightId;
        document.getElementById('deleteModal').classList.add('open');
    }

    function closeDeleteModal() {
        document.getElementById('deleteModal').classList.remove('open');
    }

    document.getElementById('deleteModal').addEventListener('click', function (e) {
        if (e.target === this) closeDeleteModal();
    });

    /* İlk yükleme */
    document.addEventListener('DOMContentLoaded', function () { applyFiltersAndRender(); });

</script>
}
