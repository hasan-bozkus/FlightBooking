    let currentStep = 1;

    function goStep(n) {
        // Mevcut adımı gizle
        document.getElementById(`step-${currentStep}`).classList.remove('active');
        document.getElementById(`step-ind-${currentStep}`).classList.remove('active');
        document.getElementById(`step-ind-${currentStep}`).classList.add('done');

        // Geri gidince "done" kaldır
        if (n < currentStep) {
            for (let i = n; i <= totalSteps; i++) {
                document.getElementById(`step-ind-${i}`)?.classList.remove('done');
            }
        }

        // Yeni adımı göster
        currentStep = n;
        document.getElementById(`step-${n}`).classList.add('active');
        document.getElementById(`step-ind-${n}`).classList.add('active');
        document.getElementById(`step-ind-${n}`).classList.remove('done');

        // Step line güncellemesi
        document.querySelectorAll('.step-line').forEach((line, idx) => {
            line.classList.toggle('done', idx < n - 1);
        });

        // Sayfayı yukarı kaydır
        window.scrollTo({ top: 0, behavior: 'smooth' });

        // Step 3 ise özet güncelle
        if (n === 3) updateSummary();
    }

    function calcDuration() {
        const depVal = document.getElementById('DepartureTime')?.value;
        const arrVal = document.getElementById('ArrivalTime')?.value;
        if (!depVal || !arrVal) return;

        const dep = new Date(depVal);
        const arr = new Date(arrVal);
        if (isNaN(dep) || isNaN(arr)) return;

        const diffMin = Math.round((arr - dep) / 60000);
        if (diffMin > 0) {
            const input = document.getElementById('durationInput');
            if (input) input.value = diffMin;
            updateDurationDisplay();
        }

        // Önizleme zamanı
        const fmt = d => d.toLocaleTimeString('tr-TR', { hour: '2-digit', minute: '2-digit' });
        document.getElementById('prev-time').textContent = `${fmt(dep)} → ${fmt(arr)}`;
    }

        if (g('sc-arr'))     g('sc-arr').textContent     = fmt(arrTime);
        if (g('sc-dur'))     g('sc-dur').textContent     = durTxt;
        if (g('sc-cap'))     g('sc-cap').textContent     = total > 0 ? `${avail}/${total}` : '—';
        if (g('sc-price'))   g('sc-price').textContent   = price > 0 ? `${symbol}${price.toLocaleString('tr-TR')}` : '—';
    }

    /* ============================================================
       FORM SUBMIT — loading state
    ============================================================ */
    document.getElementById('createFlightForm').addEventListener('submit', function () {
        const btn      = document.getElementById('submitBtn');
        const idle     = btn.querySelector('.submit-idle');
        const loading  = btn.querySelector('.submit-loading');
        idle.style.display    = 'none';
        loading.style.display = 'inline-flex';
        btn.disabled = true;
    });

    /* ── İlk yükleme ── */
    updateDurationDisplay();
</script>
